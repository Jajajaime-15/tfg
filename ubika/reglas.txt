        }

REGLAS ALBA
{
  "rules": {
    "usuarios": {
      // Permite que cualquier usuario logueado busque a otros (para invitarlos)
      ".read": "auth != null",
      "$uid": {
        ".write": "auth != null"
      }
    },
    "grupos": {
      ".read": "auth != null",
      ".write": "auth != null",
      "$grupo_id": {
        ".read": "auth != null && root.child('grupos').child($grupo_id).child('miembros').child(auth.uid).exists()",
        ".write": "auth != null && (!data.exists() || data.child('admin').val() === auth.uid)"
      }
    },
    "ubicaciones": {
      // Permite que cualquier usuario logueado lea y escriba ubicaciones. 
      // Es lo más seguro para que el rastreo en tiempo real no falle nunca.
      ".read": "auth != null",
      ".write": "auth != null"
    }
  }
}
